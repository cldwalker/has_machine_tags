      def machine_tags(name)
        conditions = if (match = match_wildcard_machine_tag(name))
          match.map {|k,v|
            sanitize_sql(["#{k} = ?", v])
          }.join(" AND ")
        else
          sanitize_sql(["name = ?", name])
        end
        find(:all, :conditions=>conditions)
      end
